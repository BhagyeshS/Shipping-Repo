--function calculates the total number of bids placed for a given shipment ID

CREATE FUNCTION CountBidsForShipment(p_ShipmentID INT)
RETURNS INT
BEGIN
    DECLARE bidCount INT;
 
    SELECT COUNT(*) INTO bidCount
    FROM Bids
    WHERE ShipmentID = p_ShipmentID;
 
    RETURN bidCount;
END //
 
DELIMITER ;
